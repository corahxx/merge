> **创建日期**: 2026-01-26

---

## 一、功能定位与目标

### 1.1 功能定位

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                          数据管理系统主界面                                   │
├────────┬────────┬────────┬────────┬────────┬────────┬────────────────────────┤
│ 数据导入│ 数据预览│ 数据分析│ 数据比对│ 数据质量│ 数据清洗│ 🆕 区域编码清洗        │
└────────┴────────┴────────┴────────┴────────┴────────┴────────────────────────┘
                                                          │
                                    ┌─────────────────────┼─────────────────────┐
                                    ▼                     ▼                     ▼
                              城市编码修复          区县编码修复           省份编码修复
                           (最后两位非00)       (与城市归属不一致)      (前2位异常)
```

### 1.2 与现有"数据清洗"功能的区别

| 功能模块 | 关注点 | 典型问题 |
|----------|--------|----------|
| **数据清洗** | 数据完整性、唯一性 | 重复记录、地址与省市区不匹配 |
| **区域编码清洗** | 编码规范性、隶属关系 | 城市代码填成区县代码、省市区隶属关系不一致 |

### 1.3 目标

1. **可视化展示**：直观呈现区域编码问题的分布和类型
2. **智能推导**：自动推导正确的编码，减少人工工作量
3. **批量修复**：支持一键修复，带进度显示和回滚能力
4. **修复追溯**：记录每次修复的详情，支持审计

## 五、核心算法

### 5.1 城市编码推导算法

```python
def derive_city_code(self, record: Dict) -> Tuple[str, str, str, float]:
    """
    推导正确的城市代码
    
    策略优先级:
    1. 直辖市规则 (置信度 1.0)
    2. 从区县代码推导 (置信度 0.95)
    3. 从当前城市代码推导 (置信度 0.85)
    4. 从地址解析 (置信度 0.70)
    """
    current_city = str(record.get('城市', '')).strip()
    district_code = str(record.get('区县', '')).strip()
    province_code = str(record.get('省份', '')).strip()
    address = str(record.get('充电站位置', '')).strip()
    
    # 获取省份前缀（优先从省份代码，否则从城市代码）
    if province_code and len(province_code) >= 2:
        province_prefix = province_code[:2]
    elif current_city and len(current_city) >= 2:
        province_prefix = current_city[:2]
    else:
        return None, None, 'NO_PREFIX', 0.0
    
    # ========== 策略1: 直辖市规则 ==========
    if province_prefix in self.DIRECT_CITY_CODES:
        code, name = self.DIRECT_CITY_CODES[province_prefix]
        return code, name, 'DIRECT_CITY', 1.0
    
    # ========== 策略2: 从区县代码推导 ==========
    if district_code and len(district_code) >= 4:
        derived = district_code[:4] + '00'
        if derived in self.city_mapping:
            # 校验: 区县前2位应与省份前2位一致
            if district_code[:2] == province_prefix:
                return derived, self.city_mapping[derived], 'FROM_DISTRICT', 0.95
    
    # ========== 策略3: 从当前城市代码推导 ==========
    if current_city and len(current_city) >= 4:
        derived = current_city[:4] + '00'
        if derived in self.city_mapping:
            # 校验: 城市前2位应与省份前2位一致
            if current_city[:2] == province_prefix:
                return derived, self.city_mapping[derived], 'FROM_CURRENT', 0.85
    
    # ========== 策略4: 从地址解析 ==========
    if address:
        parsed_city = self._parse_city_from_address(address)
        if parsed_city:
            # 在城市字典中查找
            for code, name in self.city_mapping.items():
                if name == parsed_city and code[:2] == province_prefix:
                    return code, name, 'FROM_ADDRESS', 0.70
    
    # ========== 无法自动推导 ==========
    return None, None, 'MANUAL_REQUIRED', 0.0
```

### 5.2 批量修复算法

```python
def fix_city_codes(self, fix_plan: List[Dict], 
                   batch_size: int = 500,
                   progress_callback: callable = None) -> Dict:
    """
    批量修复城市编码
    
    执行流程:
    1. 创建备份
    2. 分批更新
    3. 记录统计
    4. 清除缓存
    """
    # 1. 生成备份ID
    backup_id = f"city_fix_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
    
    # 2. 创建备份记录
    self._create_backup_records(backup_id, fix_plan)
    
    # 3. 分批执行更新
    total = len(fix_plan)
    success_count = 0
    fail_count = 0
    
    for i in range(0, total, batch_size):
        batch = fix_plan[i:i + batch_size]
        
        try:
            # 批量更新
            affected = self._execute_batch_update(batch)
            success_count += affected
            
            # 进度回调
            if progress_callback:
                progress_callback(i + len(batch), total, 
                    f"已处理 {i + len(batch):,}/{total:,}")
        except Exception as e:
            fail_count += len(batch)
            self.logger.error(f"批次 {i//batch_size + 1} 失败: {e}")
    
    # 4. 更新批次状态
    self._update_batch_status(backup_id, success_count, fail_count)
    
    # 5. 清除缓存
    self._clear_caches()
    
    return {
        'backup_id': backup_id,
        'success_count': success_count,
        'fail_count': fail_count,
        'total': total
    }
```

#### 8.2.3 地址验证算法

```python
def validate_district_with_address(record, district_mapping):
    """
    使用充电站位置验证区县代码
    
    :return: {
        'status': 'MATCH' | 'MISMATCH' | 'NO_ADDRESS' | 'PARSE_FAILED',
        'parsed_district': str or None,
        'parsed_district_code': str or None,
        'confidence': float
    }
    """
    address = str(record.get('充电站位置', '')).strip()
    current_district_name = str(record.get('区县_中文', '')).strip()
    province_code = str(record.get('省份', ''))[:2]
    
    if not address:
        return {'status': 'NO_ADDRESS', 'confidence': 0.0}
    
    # 1. 尝试从地址中提取区县名
    parsed_district = parse_district_from_address(address)
    
    if not parsed_district:
        return {'status': 'PARSE_FAILED', 'confidence': 0.0}
    
    # 2. 在字典中查找匹配的区县代码（限定在当前省份）
    matched_code = None
    for code, name in district_mapping.items():
        if code[:2] == province_code and name == parsed_district:
            matched_code = code
            break
    
    # 3. 比对结果
    if parsed_district == current_district_name:
        return {
            'status': 'MATCH',
            'parsed_district': parsed_district,
            'parsed_district_code': matched_code,
            'confidence': 0.95
        }
    else:
        return {
            'status': 'MISMATCH',
            'parsed_district': parsed_district,
            'parsed_district_code': matched_code,
            'confidence': 0.85 if matched_code else 0.70
        }


def parse_district_from_address(address):
    """
    从地址中解析区县名称
    
    支持的格式:
    - 省份 + 城市 + 区县 + 详细地址
    - 城市 + 区县 + 详细地址
    - 区县 + 详细地址
    """
    import re
    
    # 区县关键词
    district_patterns = [
        r'([^省市]+(?:区|县|市|旗|自治县|自治旗))',  # 标准区县
        r'([^省市]+(?:新区|开发区|高新区))',          # 特殊区域
    ]
    
    for pattern in district_patterns:
        match = re.search(pattern, address)
        if match:
            district = match.group(1)
            # 过滤过长的匹配（可能是误匹配）
            if len(district) <= 15:
                return district
    
    return None
```

```python
def derive_correct_district_code(record, city_mapping, district_mapping):
    """
    推导正确的区县代码（以充电站位置为准）
    
    :return: (
        fix_type: 'DISTRICT' | 'CITY' | 'BOTH' | 'NONE',
        new_district_code: str or None,
        new_district_name: str or None,
        new_city_code: str or None,
        new_city_name: str or None,
        strategy: str,
        confidence: float
    )
    """
    current_city = str(record.get('城市', '')).strip()
    current_district = str(record.get('区县', '')).strip()
    current_district_name = str(record.get('区县_中文', '')).strip()
    province_code = str(record.get('省份', '')).strip()
    address = str(record.get('充电站位置', '')).strip()
    
    province_prefix = province_code[:2] if len(province_code) >= 2 else ''
    
    # ========== 直辖市特殊处理 ==========
    DIRECT_CITIES = {'11', '12', '31', '50'}
    if province_prefix in DIRECT_CITIES:
        # 直辖市：区县代码有效的情况下，只需修复城市代码
        if current_district in district_mapping:
            city_code = f"{province_prefix}0100"
            city_name = district_mapping.get(city_code, '')
            return (
                'CITY', 
                None, None,
                city_code, city_name,
                'DIRECT_CITY_RULE', 1.0
            )
    
    # ========== 步骤1：验证当前区县代码有效性 ==========
    district_valid = current_district in district_mapping
    
    # ========== 步骤2：地址交叉验证 ==========
    if address:
        validation = validate_district_with_address(record, district_mapping)
        
        if validation['status'] == 'MATCH':
            # 地址确认区县正确，修复城市代码
            derived_city = current_district[:4] + '00'
            if derived_city in city_mapping:
                return (
                    'CITY',
                    None, None,
                    derived_city, city_mapping[derived_city],
                    'ADDRESS_CONFIRMS_DISTRICT', 0.95
                )
        
        elif validation['status'] == 'MISMATCH' and validation['parsed_district_code']:
            # 地址中的区县与当前不一致，以地址为准
            parsed_code = validation['parsed_district_code']
            parsed_name = validation['parsed_district']
            derived_city = parsed_code[:4] + '00'
            
            return (
                'BOTH',
                parsed_code, parsed_name,
                derived_city, city_mapping.get(derived_city, ''),
                'ADDRESS_OVERRIDES_DISTRICT', 0.85
            )
    
    # ========== 步骤3：无法从地址验证，基于代码规则修复 ==========
    if district_valid:
        # 区县代码有效，尝试从区县推导城市
        derived_city = current_district[:4] + '00'
        if derived_city in city_mapping:
            return (
                'CITY',
                None, None,
                derived_city, city_mapping[derived_city],
                'DERIVE_CITY_FROM_DISTRICT', 0.80
            )
    
    # ========== 无法自动修复 ==========
    return ('NONE', None, None, None, None, 'MANUAL_REQUIRED', 0.0)
```
