### 3.3 省份编码校验

**校验规则**：

| 检查项 | 规则 | 处理方式 |
|--------|------|----------|
| 格式有效 | 6位数字，后4位为0000 | 无效则尝试补齐 |
| 字典存在 | 编码在省份映射中存在 | 不存在则标记异常 |
| 名称一致 | 中文名称与编码对应 | 不一致则以编码为准修正 |

**处理逻辑**：

```python
def validate_province(province_code: str, province_name: str, 
                      province_mapping: Dict) -> Tuple[str, str, str]:
    """
    校验并修正省份编码
    返回: (修正后编码, 修正后名称, 校验状态)
    """
    code = str(province_code).strip() if province_code else ''
    name = str(province_name).strip() if province_name else ''
    
    # 格式标准化
    if len(code) == 2:
        code = code + '0000'
    elif len(code) == 6 and not code.endswith('0000'):
        code = code[:2] + '0000'
    
    # 查找正确名称
    correct_name = province_mapping.get(code)
    
    if correct_name:
        if name != correct_name:
            return code, correct_name, 'FIXED'
        return code, name, 'OK'
    else:
        # 编码无效，尝试从名称反查
        for c, n in province_mapping.items():
            if n == name:
                return c, n, 'CODE_FIXED'
        return code, name, 'INVALID'
```

### 3.4 城市编码校验（重点）

参考 **城市编码修复可行性分析.md** 的分级修复策略：

**问题背景**：
- 约2.47%的记录城市编码异常（末尾非00）
- 96.1%的异常是错误地填入了区县编码
- 直辖市异常比例较高（上海12%、重庆8.8%）

**分级修复策略**：

```

│                      城市编码修复流程                            │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  Step 1: 判断是否需要修复                                       │
│  └─ 城市编码最后两位 == '00' → 无需修复，跳过                   │
│  └─ 城市编码最后两位 != '00' → 需要修复，继续                   │
│                                                                 │
│  Step 2: 直辖市特殊处理（置信度：100%）                         │
│  └─ 判断省份是否为直辖市（11/12/31/50）                         │

│     └─ 失败 → 进入Step 4                                        │
│                                                                 │
│  Step 4: 从当前城市编码推导（置信度：85%）                      │
│  └─ 推导：城市编码 = 当前编码前4位 + '00'                       │
│  └─ 校验：推导结果在城市字典中存在                              │
│  └─ 校验：推导结果前2位与省份编码前2位一致                      │
```

**向量化实现**：

```python
def fix_city_codes_vectorized(df: pd.DataFrame, 
                               city_mapping: Dict,
                               district_mapping: Dict) -> pd.DataFrame:
    """
    向量化城市编码修复
    """
    df = df.copy()
    
    # 直辖市固定编码
    DIRECT_CITY = {
        '11': '110100', '12': '120100', 
        '31': '310100', '50': '500100'
    }
    
    # 1. 标准化编码格式
    df['_city_code'] = df['城市'].astype(str).str.strip()
    df['_district_code'] = df['区县'].astype(str).str.strip()
    df['_province_prefix'] = df['省份'].astype(str).str[:2]
    
    # 2. 标记需要修复的记录
    df['_needs_fix'] = ~df['_city_code'].str.endswith('00')
    
    # 3. 直辖市处理
    is_direct = df['_province_prefix'].isin(DIRECT_CITY.keys()) & df['_needs_fix']
    df.loc[is_direct, '城市'] = df.loc[is_direct, '_province_prefix'].map(DIRECT_CITY)
    df.loc[is_direct, '_needs_fix'] = False
    
    # 4. 从区县编码推导
    df['_derived_from_district'] = df['_district_code'].str[:4] + '00'
    can_derive_from_district = (
        df['_needs_fix'] & 
        (df['_district_code'].str.len() >= 4) &
        df['_derived_from_district'].isin(city_mapping.keys()) &
        (df['_derived_from_district'].str[:2] == df['_province_prefix'])
    )
    df.loc[can_derive_from_district, '城市'] = df.loc[can_derive_from_district, '_derived_from_district']
    df.loc[can_derive_from_district, '_needs_fix'] = False
    
    # 5. 从当前编码推导
    df['_derived_from_current'] = df['_city_code'].str[:4] + '00'
    can_derive_from_current = (
        df['_needs_fix'] & 
        df['_derived_from_current'].isin(city_mapping.keys()) &
        (df['_derived_from_current'].str[:2] == df['_province_prefix'])
    )
    df.loc[can_derive_from_current, '城市'] = df.loc[can_derive_from_current, '_derived_from_current']
    df.loc[can_derive_from_current, '_needs_fix'] = False
    
    # 6. 仍需修复的标记为异常
    df.loc[df['_needs_fix'], 'data_status'] = 5  # 待人工处理
    
    # 7. 清理临时列
    df = df.drop(columns=[c for c in df.columns if c.startswith('_')])
    
    # 8. 更新城市中文名称
    df['城市_中文'] = df['城市'].map(city_mapping).fillna(df['城市_中文'])
    
    return df
```

### 3.5 区县编码校验

**校验规则**：

| 检查项 | 规则 | 处理方式 |
|--------|------|----------|
| 格式有效 | 6位数字 | 不足则左补0 |
| 字典存在 | 编码在区县映射中存在 | 不存在则标记异常 |
| 隶属一致 | 前4位与城市编码前4位一致 | 不一致则标记警告 |
| 名称一致 | 中文名称与编码对应 | 不一致则以编码为准 |

**向量化实现**：

```python
def fix_district_codes_vectorized(df: pd.DataFrame,
                                   district_mapping: Dict) -> pd.DataFrame:
    """
    向量化区县编码修复
    """
    df = df.copy()
    
    # 1. 标准化编码格式（6位，不足左补0）
    df['区县'] = df['区县'].astype(str).str.strip().str.zfill(6)
    
    # 2. 校验编码是否在字典中存在
    df['_district_valid'] = df['区县'].isin(district_mapping.keys())
    
    # 3. 校验隶属关系（区县前4位 = 城市前4位）
    df['_city_prefix'] = df['城市'].astype(str).str[:4]
    df['_district_prefix'] = df['区县'].str[:4]
    df['_belongs_to_city'] = df['_city_prefix'] == df['_district_prefix']
    
    # 4. 无效编码标记
    df.loc[~df['_district_valid'], 'data_status'] = 6  # 区县编码无效
    
    # 5. 隶属关系不一致标记警告（但不阻止导入）
    mismatch = df['_district_valid'] & ~df['_belongs_to_city']
    # 可选：记录警告日志
    
    # 6. 更新区县中文名称
    df['区县_中文'] = df['区县'].map(district_mapping).fillna(df['区县_中文'])
    
    # 7. 清理临时列
    df = df.drop(columns=[c for c in df.columns if c.startswith('_')])
    
    return df
```

### 3.6 三级联动校验

**最终校验：确保省-市-区三级编码隶属一致**

```python
def validate_region_hierarchy(df: pd.DataFrame) -> pd.DataFrame:
    """
    校验省市区三级编码的隶属关系
    """
    df = df.copy()
    
    # 提取编码前缀
    df['_prov_prefix'] = df['省份'].astype(str).str[:2]
    df['_city_prefix'] = df['城市'].astype(str).str[:2]
    df['_dist_prefix'] = df['区县'].astype(str).str[:2]
    
    df['_city_to_dist'] = df['城市'].astype(str).str[:4]
    df['_dist_city_part'] = df['区县'].astype(str).str[:4]
    
    # 校验隶属关系
    # 1. 省份-城市一致性
    prov_city_mismatch = df['_prov_prefix'] != df['_city_prefix']
    
    # 2. 城市-区县一致性  
    city_dist_mismatch = df['_city_to_dist'] != df['_dist_city_part']
    
    # 标记不一致的记录
    df.loc[prov_city_mismatch | city_dist_mismatch, 'data_status'] = 7  # 区域隶属异常
    
    # 清理
    df = df.drop(columns=[c for c in df.columns if c.startswith('_')])
    
    return df
```

### 4.3 向量化实现

```python
import pandas as pd
import re
from datetime import datetime

def standardize_date_vectorized(df: pd.DataFrame, 
                                  date_column: str = '充电站投入使用时间') -> pd.DataFrame:
    """
    向量化日期格式标准化
    
    处理策略：
    1. 时间戳格式 → 截取日期
    2. 多种日期格式 → 统一为 DATE 类型
    3. 无法解析 → NULL
    """
    df = df.copy()
    
    if date_column not in df.columns:
        return df
    
    # 保存原始值用于日志
    df['_original_date'] = df[date_column]
    
    # 转为字符串处理
    df[date_column] = df[date_column].astype(str).str.strip()
    
    # 空值处理
    empty_mask = (
        df[date_column].isna() | 
        (df[date_column] == '') | 
        (df[date_column] == 'nan') |
        (df[date_column] == 'None')
    )
    df.loc[empty_mask, date_column] = None
    
    # 非空值处理
    non_empty = ~empty_mask
    
    def parse_date(val):
        """解析单个日期值"""
        if pd.isna(val) or val in ('', 'nan', 'None'):
            return None
        
        val = str(val).strip()
        
        # 如果包含时间部分，先截取日期
        if ' ' in val:
            val = val.split(' ')[0]
        if 'T' in val:
            val = val.split('T')[0]
        
        # 定义多种可能的格式
        formats = [
            '%Y/%m/%d',
            '%Y-%m-%d',
            '%Y%m%d',
            '%Y年%m月%d日',
            '%Y.%m.%d',
            '%d/%m/%Y',
            '%m/%d/%Y',
        ]
        
        for fmt in formats:
            try:
                dt = datetime.strptime(val, fmt)
                # 有效性校验
                if 1990 <= dt.year <= datetime.now().year + 1:
                    return dt.date()
            except ValueError:
                continue
        
        # 尝试解析8位纯数字
        if re.match(r'^\d{8}$', val):
            try:
                dt = datetime.strptime(val, '%Y%m%d')
                if 1990 <= dt.year <= datetime.now().year + 1:
                    return dt.date()
            except:
                pass
        
        # 无法解析
        return None
    
    # 应用解析函数
    df.loc[non_empty, date_column] = df.loc[non_empty, date_column].apply(parse_date)
    
    # 统计转换失败的记录
    failed_mask = non_empty & df[date_column].isna()
    failed_count = failed_mask.sum()
    
    if failed_count > 0:
        # 记录无法转换的原始值（用于日志）
        failed_values = df.loc[failed_mask, '_original_date'].unique()[:10]
        print(f"⚠️ 日期转换: {failed_count}条无法解析，示例: {list(failed_values)}")
    
    # 清理临时列
    df = df.drop(columns=['_original_date'])
    
    return df
```

```python
# data/data_cleaner.py 改造

class DataCleaner:
    
    def clean(self, df: pd.DataFrame, field_mapping: Optional[Dict] = None) -> pd.DataFrame:
        """
        执行完整的数据清洗流程（改造版）
        """
        self.cleaning_stats['rows_before'] = len(df)
        df_cleaned = df.copy()
        
        # 1. 生成UID主键
        df_cleaned = self._generate_uids(df_cleaned)
        
        # 2. 字段映射
        if field_mapping:
            df_cleaned = self._map_columns(df_cleaned, field_mapping)
        
        # 3. 去除空白
        df_cleaned = self._strip_whitespace(df_cleaned)
        
        # 4. 标准化字段（运营商名称等）
        df_cleaned = self._normalize_fields(df_cleaned)
        
        # ========== 5. 区域编码严格校验（改造重点） ==========
        df_cleaned = self._fix_region_codes_strict(df_cleaned)
        
        # ========== 6. 日期格式标准化（新增） ==========
        df_cleaned = self._standardize_dates(df_cleaned)
        
        # 7. 类型转换
        schema = self._get_schema()
        if schema and self.engine:
            df_cleaned = self._convert_to_database_types(df_cleaned, schema)
        
        # 8. 截断超长字符串
        if schema:
            df_cleaned = self._truncate_string_fields(df_cleaned, schema)
        
        # 9. 处理空值
        df_cleaned = self._handle_nulls(df_cleaned)
        
        # 10. 去重（基于充电桩编号，可选）
        df_cleaned = self._remove_duplicates(df_cleaned)
        
        # 11. 数据验证
        df_cleaned = self._validate_data(df_cleaned)
        
        self.cleaning_stats['rows_after'] = len(df_cleaned)
        
        if self.verbose:
            self._print_stats()
        
        return df_cleaned
    
    def _generate_uids(self, df: pd.DataFrame) -> pd.DataFrame:
        """生成UID主键（向量化）"""
        import uuid
        import numpy as np
        
        if 'UID' not in df.columns or df['UID'].isna().all():
            df['UID'] = np.array([str(uuid.uuid4()) for _ in range(len(df))])
            if self.verbose:
                print(f"✅ 生成了 {len(df)} 个UID")
        return df
    
    def _fix_region_codes_strict(self, df: pd.DataFrame) -> pd.DataFrame:
        """
        区域编码严格校验（向量化）
        包含：省份校验、城市修复、区县校验、三级联动校验
        """
        try:
            converter = get_converter()
            province_map = converter.provinces
            city_map = converter.cities
            district_map = converter.districts
        except Exception as e:
            if self.verbose:
                print(f"⚠️ 区域转换器加载失败: {e}")
            return df
        
        df_fixed = df.copy()
        stats = {'province_fixed': 0, 'city_fixed': 0, 'district_fixed': 0}
        
        # 直辖市固定编码
        DIRECT_CITY = {'11': '110100', '12': '120100', '31': '310100', '50': '500100'}
        
        # ========== 省份校验 ==========
        if '省份' in df.columns:
            df_fixed['省份'] = df_fixed['省份'].astype(str).str.strip()
            # 标准化为6位
            mask_short = df_fixed['省份'].str.len() == 2
            df_fixed.loc[mask_short, '省份'] = df_fixed.loc[mask_short, '省份'] + '0000'
            
            # 更新中文名称
            if '省份_中文' in df.columns:
                new_names = df_fixed['省份'].map(province_map)
                changed = (new_names.notna()) & (df_fixed['省份_中文'] != new_names)
                stats['province_fixed'] = changed.sum()
                df_fixed.loc[new_names.notna(), '省份_中文'] = new_names[new_names.notna()]
        
        # ========== 城市编码修复（分级策略） ==========
        if '城市' in df.columns:
            df_fixed['城市'] = df_fixed['城市'].astype(str).str.strip()
            df_fixed['_province_prefix'] = df_fixed['省份'].astype(str).str[:2]
            
            # 需要修复的记录
            needs_fix = ~df_fixed['城市'].str.endswith('00')
            
            # 策略1: 直辖市
            is_direct = df_fixed['_province_prefix'].isin(DIRECT_CITY.keys()) & needs_fix
            df_fixed.loc[is_direct, '城市'] = df_fixed.loc[is_direct, '_province_prefix'].map(DIRECT_CITY)
            needs_fix = needs_fix & ~is_direct
            
            # 策略2: 从区县推导
            if '区县' in df.columns:
                df_fixed['_district_code'] = df_fixed['区县'].astype(str).str.strip().str.zfill(6)
                df_fixed['_derived_city'] = df_fixed['_district_code'].str[:4] + '00'
                
                can_derive = (
                    needs_fix & 
                    (df_fixed['_district_code'].str.len() >= 4) &
                    df_fixed['_derived_city'].isin(city_map.keys()) &
                    (df_fixed['_derived_city'].str[:2] == df_fixed['_province_prefix'])
                )
                df_fixed.loc[can_derive, '城市'] = df_fixed.loc[can_derive, '_derived_city']
                needs_fix = needs_fix & ~can_derive
            
            # 策略3: 从当前编码推导
            df_fixed['_current_derived'] = df_fixed['城市'].str[:4] + '00'
            can_derive_current = (
                needs_fix &
                df_fixed['_current_derived'].isin(city_map.keys()) &
                (df_fixed['_current_derived'].str[:2] == df_fixed['_province_prefix'])
            )
            df_fixed.loc[can_derive_current, '城市'] = df_fixed.loc[can_derive_current, '_current_derived']
            needs_fix = needs_fix & ~can_derive_current
            
            # 仍无法修复的标记
            if 'data_status' in df_fixed.columns:
                df_fixed.loc[needs_fix, 'data_status'] = 5  # 待人工处理
            
            # 更新城市中文名称
            if '城市_中文' in df.columns:
                new_city_names = df_fixed['城市'].map(city_map)
                changed = (new_city_names.notna()) & (df_fixed['城市_中文'] != new_city_names)
                stats['city_fixed'] = changed.sum()
                df_fixed.loc[new_city_names.notna(), '城市_中文'] = new_city_names[new_city_names.notna()]
        
        # ========== 区县校验 ==========
        if '区县' in df.columns:
            df_fixed['区县'] = df_fixed['区县'].astype(str).str.strip().str.zfill(6)
            
            # 更新区县中文名称
            if '区县_中文' in df.columns:
                new_district_names = df_fixed['区县'].map(district_map)
                changed = (new_district_names.notna()) & (df_fixed['区县_中文'] != new_district_names)
                stats['district_fixed'] = changed.sum()
                df_fixed.loc[new_district_names.notna(), '区县_中文'] = new_district_names[new_district_names.notna()]
        
        # 清理临时列
        temp_cols = [c for c in df_fixed.columns if c.startswith('_')]
        df_fixed = df_fixed.drop(columns=temp_cols, errors='ignore')
        
        # 更新统计
        self.cleaning_stats['region_fixed'] = stats
        total_fixed = sum(stats.values())
        if self.verbose and total_fixed > 0:
            print(f"✅ 区域编码校验: 省份 {stats['province_fixed']}, "
                  f"城市 {stats['city_fixed']}, 区县 {stats['district_fixed']}")
        
        return df_fixed
    
    def _standardize_dates(self, df: pd.DataFrame) -> pd.DataFrame:
        """日期格式标准化"""
        date_columns = ['充电站投入使用时间', '充电桩生产日期']
        
        for col in date_columns:
            if col in df.columns:
                df = standardize_date_vectorized(df, col)
        
        return df
```
