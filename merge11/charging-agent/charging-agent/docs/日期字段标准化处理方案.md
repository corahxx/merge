# 充电站投入使用时间字段标准化处理方案

> **创建日期**: 2026-01-26  
> **问题描述**: 将 `充电站投入使用时间` 字段从 VARCHAR 转换为 DATE 类型  
> **数据量**: 3,655,982 条活跃记录

## 一、数据现状分析

### 1.1 基本统计

| 指标 | 数值 |
|------|------|
| 总活跃记录数 | 3,655,982 |
| 空值/NULL记录 | 4,902 |
| 有效日期记录 | 3,651,080 |

### 1.2 格式分布

通过采样分析，识别出以下主要日期格式：

| 格式模式 | 示例 | 预估记录数 | 占比 |
|----------|------|-----------|------|

| `YYYY/M/D H:MM` | 2023/7/11 5:02 | ~200 | <0.1% |

### 1.3 异常情况

- 部分记录包含时间戳，需要截取日期部分
- 存在美式日期格式（月/日/年）
- 可能存在不可见字符或空格

## 三、SQL 处理脚本

### 3.1 Step 1: 准备工作

```sql
WHERE is_active = 1

-- ==========================================
-- Step 1.2: 添加标准化临时字段
-- ==========================================
ALTER TABLE evdata 
ADD COLUMN 投入使用日期_标准化 VARCHAR(10) NULL AFTER 充电站投入使用时间;

-- 添加最终的 DATE 类型字段
ALTER TABLE evdata 
ADD COLUMN 投入使用日期 DATE NULL AFTER 投入使用日期_标准化;
```

### 3.2 Step 2: 格式转换（分批处理）

```sql
-- ==========================================
-- Step 2.1: 处理 YYYY/MM/DD 格式（最标准，直接替换）
-- 匹配: 2024/12/26, 2023/10/15 等
-- ==========================================
UPDATE evdata
SET 投入使用日期_标准化 = REPLACE(
    SUBSTRING_INDEX(充电站投入使用时间, ' ', 1),  -- 去除可能的时间部分
    '/', '-'
)
WHERE is_active = 1
AND 充电站投入使用时间 REGEXP '^[0-9]{4}/[0-9]{2}/[0-9]{2}'
AND 投入使用日期_标准化 IS NULL;

-- 查看处理结果
SELECT '格式1: YYYY/MM/DD' AS format_type, COUNT(*) AS processed
FROM evdata WHERE 投入使用日期_标准化 IS NOT NULL;

-- ==========================================
-- Step 2.2: 处理 YYYY/M/DD 格式（月份需补零）
-- 匹配: 2025/7/23, 2023/1/15 等
-- ==========================================
UPDATE evdata
SET 投入使用日期_标准化 = CONCAT(
    SUBSTRING_INDEX(充电站投入使用时间, '/', 1), '-',
    LPAD(
        SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, '/', 2), '/', -1),
        2, '0'
    ), '-',
    LPAD(
        SUBSTRING_INDEX(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, ' ', 1), '/', 3), '/', -1),
        2, '0'
    )
)
WHERE is_active = 1
AND 充电站投入使用时间 REGEXP '^[0-9]{4}/[0-9]{1,2}/[0-9]{1,2}'
AND 投入使用日期_标准化 IS NULL;

-- 查看处理结果
SELECT '格式2: YYYY/M/D' AS format_type, COUNT(*) AS processed
FROM evdata WHERE 投入使用日期_标准化 IS NOT NULL;

-- ==========================================
-- Step 2.3: 处理 YYYY-MM-DD HH:MM:SS 格式
-- 匹配: 2022-11-20 06:02:03 等
-- ==========================================
UPDATE evdata
SET 投入使用日期_标准化 = SUBSTRING(充电站投入使用时间, 1, 10)
WHERE is_active = 1
AND 充电站投入使用时间 REGEXP '^[0-9]{4}-[0-9]{2}-[0-9]{2}'
AND 投入使用日期_标准化 IS NULL;

-- 查看处理结果
SELECT '格式3: YYYY-MM-DD HH:MM:SS' AS format_type, COUNT(*) AS processed
FROM evdata WHERE 投入使用日期_标准化 IS NOT NULL;

-- ==========================================
-- Step 2.4: 处理美式格式 MM DD YYYY HH:MMAM
-- 匹配: 12 13 2022 12:00AM, 02 10 2023 12:00AM 等
-- ==========================================
UPDATE evdata
SET 投入使用日期_标准化 = CONCAT(
    -- 年份（第3部分）
    TRIM(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, ' ', 3), ' ', -1)), '-',
    -- 月份（第1部分，补零）
    LPAD(TRIM(SUBSTRING_INDEX(充电站投入使用时间, ' ', 1)), 2, '0'), '-',
    -- 日期（第2部分，补零）
    LPAD(TRIM(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, ' ', 2), ' ', -1)), 2, '0')
)
WHERE is_active = 1
AND 充电站投入使用时间 REGEXP '^[0-9]{1,2} +[0-9]{1,2} +[0-9]{4}'
AND 投入使用日期_标准化 IS NULL;

-- 查看处理结果
SELECT '格式4: MM DD YYYY' AS format_type, COUNT(*) AS processed
FROM evdata WHERE 投入使用日期_标准化 IS NOT NULL;

-- ==========================================
-- Step 2.5: 处理 YYYY年MM月DD日 格式（如果存在）
-- ==========================================
UPDATE evdata
SET 投入使用日期_标准化 = CONCAT(
    SUBSTRING_INDEX(充电站投入使用时间, '年', 1), '-',
    LPAD(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, '月', 1), '年', -1), 2, '0'), '-',
    LPAD(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, '日', 1), '月', -1), 2, '0')
)
WHERE is_active = 1
AND 充电站投入使用时间 LIKE '%年%月%日%'
AND 投入使用日期_标准化 IS NULL;

-- ==========================================
-- Step 2.6: 处理剩余的 YYYY-M-D 格式
-- ==========================================
UPDATE evdata
SET 投入使用日期_标准化 = CONCAT(
    SUBSTRING_INDEX(充电站投入使用时间, '-', 1), '-',
    LPAD(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, '-', 2), '-', -1), 2, '0'), '-',
    LPAD(SUBSTRING_INDEX(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, ' ', 1), '-', 3), '-', -1), 2, '0')
)
WHERE is_active = 1
AND 充电站投入使用时间 REGEXP '^[0-9]{4}-[0-9]{1,2}-[0-9]{1,2}'
AND 投入使用日期_标准化 IS NULL;
```

```sql
FROM evdata 
WHERE is_active = 1

    CAST(SUBSTRING(投入使用日期_标准化, 1, 4) AS UNSIGNED) < 1990

WHERE is_active = 1

AND 投入使用日期_标准化 REGEXP '^[0-9]{4}-[0-9]{2}-[0-9]{2}$';

    COUNT(*) AS total,

    SUM(CASE WHEN 投入使用日期 IS NULL AND 投入使用日期_标准化 IS NOT NULL THEN 1 ELSE 0 END) AS failed
FROM evdata WHERE is_active = 1;
```

## 四、分批执行脚本（适用于大数据量）

考虑到 360万+ 条记录，建议使用分批处理避免锁表：

```sql
-- ==========================================
-- 分批处理模板（每批 50000 条）
-- ==========================================

-- 创建处理状态追踪表
CREATE TABLE IF NOT EXISTS date_conversion_log (
    batch_id INT AUTO_INCREMENT PRIMARY KEY,
    start_uid VARCHAR(50),
    end_uid VARCHAR(50),
    processed_count INT,
    started_at DATETIME,
    completed_at DATETIME,
    status VARCHAR(20)
);

-- 分批处理存储过程
DELIMITER //

CREATE PROCEDURE BatchConvertDates(IN batch_size INT)
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE v_min_uid VARCHAR(50);
    DECLARE v_max_uid VARCHAR(50);
    DECLARE v_count INT DEFAULT 0;
    DECLARE v_batch INT DEFAULT 0;
    
    -- 获取总记录数
    SELECT COUNT(*) INTO @total FROM evdata 
    WHERE is_active = 1 
    AND 充电站投入使用时间 IS NOT NULL 
    AND 充电站投入使用时间 != ''
    AND 投入使用日期_标准化 IS NULL;
    
    SELECT CONCAT('Total records to process: ', @total) AS info;
    
    WHILE @total > 0 DO
        SET v_batch = v_batch + 1;
        
        -- 获取本批次的UID范围
        SELECT MIN(UID), MAX(UID), COUNT(*) 
        INTO v_min_uid, v_max_uid, v_count
        FROM (
            SELECT UID FROM evdata 
            WHERE is_active = 1 
            AND 充电站投入使用时间 IS NOT NULL 
            AND 充电站投入使用时间 != ''
            AND 投入使用日期_标准化 IS NULL
            LIMIT batch_size
        ) t;
        
        -- 记录开始
        INSERT INTO date_conversion_log (start_uid, end_uid, started_at, status)
        VALUES (v_min_uid, v_max_uid, NOW(), 'processing');
        SET @log_id = LAST_INSERT_ID();
        
        -- 执行转换（YYYY/MM/DD 格式）
        UPDATE evdata
        SET 投入使用日期_标准化 = REPLACE(
            SUBSTRING_INDEX(充电站投入使用时间, ' ', 1), '/', '-'
        )
        WHERE is_active = 1
        AND 充电站投入使用时间 REGEXP '^[0-9]{4}/[0-9]{2}/[0-9]{2}'
        AND 投入使用日期_标准化 IS NULL
        AND UID >= v_min_uid AND UID <= v_max_uid;
        
        -- 执行转换（YYYY/M/D 格式）
        UPDATE evdata
        SET 投入使用日期_标准化 = CONCAT(
            SUBSTRING_INDEX(充电站投入使用时间, '/', 1), '-',
            LPAD(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, '/', 2), '/', -1), 2, '0'), '-',
            LPAD(SUBSTRING_INDEX(SUBSTRING_INDEX(SUBSTRING_INDEX(充电站投入使用时间, ' ', 1), '/', 3), '/', -1), 2, '0')
        )
        WHERE is_active = 1
        AND 充电站投入使用时间 REGEXP '^[0-9]{4}/[0-9]{1,2}/[0-9]{1,2}'
        AND 投入使用日期_标准化 IS NULL
        AND UID >= v_min_uid AND UID <= v_max_uid;
        
        -- 执行转换（YYYY-MM-DD HH:MM:SS 格式）
        UPDATE evdata
        SET 投入使用日期_标准化 = SUBSTRING(充电站投入使用时间, 1, 10)
        WHERE is_active = 1
        AND 充电站投入使用时间 REGEXP '^[0-9]{4}-[0-9]{2}-[0-9]{2}'
        AND 投入使用日期_标准化 IS NULL
        AND UID >= v_min_uid AND UID <= v_max_uid;
        
        -- 更新日志
        UPDATE date_conversion_log 
        SET processed_count = v_count, completed_at = NOW(), status = 'completed'
        WHERE batch_id = @log_id;
        
        -- 输出进度
        SELECT CONCAT('Batch ', v_batch, ': processed ', v_count, ' records') AS progress;
        
        -- 更新剩余数量
        SET @total = @total - v_count;
        
        -- 短暂休息避免锁表
        DO SLEEP(0.1);
    END WHILE;
    
    SELECT 'All batches completed!' AS result;
END //

DELIMITER ;
```
