## 一、当前问题分析

### 1.1 数据预览性能问题

**当前代码** (`handlers/data_preview_handler.py`):
```python
query = text(f"SELECT * FROM `{self.table_name}` LIMIT {limit}")
```

**问题**：
| 问题点 | 影响 | 严重程度 |
|--------|------|----------|
| 使用 `SELECT *` | 查询所有45+个字段，大量无用数据传输 | 🔴 严重 |
| 默认无上限 | 可能返回过多数据导致内存问题 | 🟡 中等 |
| 包含空值字段 | 增加数据量，降低可读性 | 🟠 低 |

### 1.2 统计分析性能问题

**当前代码** (`data/data_analyzer.py`):
```python
query = f"SELECT * FROM `{self.table_name}`"
if where_conditions:
    query += " WHERE " + " AND ".join(where_conditions)
if use_limit:
    query += f" LIMIT {limit}"

df = pd.read_sql(query, self.engine)
```

**问题**：
| 问题点 | 影响 | 严重程度 |
|--------|------|----------|
| 全字段查询 | 统计只需5-6个字段，却查询45+个 | 🔴 严重 |
| 先查询后聚合 | 将大量数据加载到内存再做统计 | 🔴 严重 |
| 无缓存机制 | 相同查询反复执行 | 🟡 中等 |

### 1.3 性能影响估算

假设当前数据量：**409万条记录**，每行约45个字段

| 场景 | 当前耗时 | 数据传输量 |
|------|----------|-----------|
| 数据预览(100条) | ~1-2秒 | ~100KB |
| 统计分析(1万条) | ~5-10秒 | ~10MB |
| 统计分析(全量) | ~30-60秒 | ~400MB+ |

```python
# handlers/data_preview_handler.py

class DataPreviewHandler:
    # 核心预览字段（不包含可空的次要字段）
    PREVIEW_CORE_FIELDS = [
        '运营商名称', '省份_中文', '城市_中文', '区县_中文',
        '充电站名称', '充电桩类型_转换', '额定功率',
        '充电桩编号', '所属充电站编号', 'UID'
    ]
    
    # 最大预览行数
    MAX_PREVIEW_LIMIT = 1000
    DEFAULT_PREVIEW_LIMIT = 100
    
    def preview_data(self, limit: int = None, show_all_fields: bool = False) -> Dict:
        """
        预览数据（优化版）
        :param limit: 预览行数（默认100，最大1000）
        :param show_all_fields: 是否显示所有字段
        """
        # 限制最大行数
        if limit is None:
            limit = self.DEFAULT_PREVIEW_LIMIT
        limit = min(limit, self.MAX_PREVIEW_LIMIT)
        
        # 选择字段
        if show_all_fields:
            fields = '*'
        else:
            fields = ', '.join([f'`{f}`' for f in self.PREVIEW_CORE_FIELDS])
        
        query = text(f"SELECT {fields} FROM `{self.table_name}` LIMIT {limit}")
        df = pd.read_sql(query, self.engine)
        
        return {
            'success': True,
            'data': df,
            'row_count': len(df),
            'column_count': len(df.columns),
            'is_limited': limit == self.MAX_PREVIEW_LIMIT
        }
```

### 2.2 统计分析优化

#### 方案设计

**核心思路**：用SQL聚合代替Python聚合，只查询必要字段

| 统计类型 | 当前查询 | 优化查询 |
|----------|----------|----------|
| 按运营商统计 | SELECT * | SELECT 运营商名称, COUNT(*) GROUP BY |
| 按区域统计 | SELECT * | SELECT 省份_中文, COUNT(*) GROUP BY |
| 按类型统计 | SELECT * | SELECT 充电桩类型_转换, COUNT(*) GROUP BY |
| 按功率统计 | SELECT * | SELECT 额定功率, COUNT(*) GROUP BY |

#### 优化后的SQL示例

```sql
-- 优化前：全字段查询后在Python中聚合
SELECT * FROM evdata WHERE 省份_中文 = '广东省';
-- 返回数十万条完整记录，Python中value_counts()

-- 优化后：SQL直接聚合
SELECT 
    运营商名称,
    COUNT(*) as count
FROM evdata 
WHERE 省份_中文 = '广东省'
GROUP BY 运营商名称
ORDER BY count DESC;
-- 只返回几十条聚合结果
```

#### 代码修改

```python
# data/data_analyzer.py

def get_statistics_optimized(self, analysis_type: str, 
                             region_filter: Dict = None, 
                             operator_filter: List[str] = None,
                             ...) -> Dict:
    """
    优化版统计分析：使用SQL聚合代替全表查询
    """
    # 构建WHERE条件
    where_conditions = self._build_where_conditions(
        region_filter, operator_filter, power_filter, ...
    )
    where_clause = f"WHERE {' AND '.join(where_conditions)}" if where_conditions else ""
    
    stats = {'total_records': 0, 'basic_stats': {}}
    
    # 1. 总数查询（只COUNT不取数据）
    count_query = f"SELECT COUNT(*) as total FROM `{self.table_name}` {where_clause}"
    stats['total_records'] = pd.read_sql(count_query, self.engine).iloc[0]['total']
    
    # 2. 按运营商聚合
    operator_query = f"""
        SELECT 运营商名称, COUNT(*) as count 
        FROM `{self.table_name}` {where_clause}
        GROUP BY 运营商名称 
        ORDER BY count DESC
    """
    operator_df = pd.read_sql(operator_query, self.engine)
    stats['basic_stats']['by_operator'] = dict(zip(
        operator_df['运营商名称'], operator_df['count']
    ))
    
    # 3. 按区域聚合（根据查询级别动态选择字段）
    location_field = self._determine_location_field(region_filter)
    location_query = f"""
        SELECT `{location_field}`, COUNT(*) as count 
        FROM `{self.table_name}` {where_clause}
        GROUP BY `{location_field}` 
        ORDER BY count DESC
    """
    location_df = pd.read_sql(location_query, self.engine)
    stats['basic_stats']['by_location'] = dict(zip(
        location_df[location_field], location_df['count']
    ))
    
    # 4. 按类型聚合
    type_query = f"""
        SELECT 充电桩类型_转换, COUNT(*) as count 
        FROM `{self.table_name}` {where_clause}
        GROUP BY 充电桩类型_转换
    """
    type_df = pd.read_sql(type_query, self.engine)
    stats['basic_stats']['by_type'] = dict(zip(
        type_df['充电桩类型_转换'], type_df['count']
    ))
    
    return stats
```

```python
def preview_data_paginated(self, page: int = 1, page_size: int = 100) -> Dict:
    """
    分页预览数据
    """
    offset = (page - 1) * page_size
    query = text(f"""
        SELECT {fields}
```

#### 2.3.4 异步查询（高级）

```python
# 对于复杂的多维统计，可以并行执行
import asyncio
from concurrent.futures import ThreadPoolExecutor

async def get_all_statistics_async(self, ...):
    """异步并行获取多个统计结果"""
    with ThreadPoolExecutor(max_workers=4) as executor:
        loop = asyncio.get_event_loop()
        
        # 并行执行多个查询
        tasks = [
            loop.run_in_executor(executor, self.get_operator_stats),
            loop.run_in_executor(executor, self.get_region_stats),
            loop.run_in_executor(executor, self.get_type_stats),
            loop.run_in_executor(executor, self.get_power_stats),
        ]
        
        results = await asyncio.gather(*tasks)
        
    return {
        'by_operator': results[0],
        'by_region': results[1],
        'by_type': results[2],
        'by_power': results[3],
    }
```
